%% LaTeX Layout for CV
%% Michael Malick

\documentclass[12pt]{article}
\usepackage[utf8]{inputenc}
\usepackage{geometry}
\geometry{verbose, letterpaper,
  tmargin = 2.54cm, bmargin = 2.54cm,
  lmargin = 2.64cm, rmargin = 2.64cm}

% setup description list
\usepackage{enumitem}
\setlist[description]{font=\normalfont\small,leftmargin=50pt,style=nextline, topsep=1em,itemsep=1pt}

% remove section numbers
\setcounter{secnumdepth}{0}

% make header larger
\usepackage{titlesec}
\titleformat*{\section}{\LARGE\bfseries}
\titleformat*{\subsection}{\Large\bfseries}
% \titlespacing*{\section}{0pt}{10pt}{2pt}
% \titlespacing*{\subsection}{0pt}{10pt}{2pt}

% needed to fix \tightlist definition used in pandoc
\providecommand{\tightlist}{ }

% header and footer
\usepackage{fancyhdr}
\setlength{\headheight}{15.2pt}
\renewcommand{\headrulewidth}{0pt} % remove hrule
\renewcommand{\footrulewidth}{0.2pt}
\fancyhf{} % clears default headers/footers
\pagestyle{fancy}
\fancypagestyle{firststyle} {
  \fancyhf{}
  \rfoot{\textit{\thepage}}
  \lfoot{\textit{\headerdate\today}}
}
\rhead{\textit{Michael J. Malick}}
\lfoot{\textit{\headerdate\today}}
\rfoot{\textit{\thepage}}

% use and color hyperlinks
\usepackage[usenames,dvipsnames]{color}
\usepackage[colorlinks=true,urlcolor=MidnightBlue]{hyperref}

% change font of hyperlinks
\usepackage{url}
\urlstyle{rm}

% change date format for footer
\usepackage{datetime}
\newdateformat{headerdate}{\monthname[\THEMONTH] \THEYEAR}

% no indenting
\setlength{\parindent}{0pt}

%% Libertine font
%% http://mirror.utexas.edu/ctan/fonts/newtx/doc/newtxdoc.pdf (pg. 9)
\usepackage[lining,semibold]{libertine} % a bit lighter than Times--no osf in math
\usepackage[T1]{fontenc} % best for Western European languages
\usepackage{textcomp} % required to get special symbols
\usepackage[varqu,varl]{inconsolata}% a typewriter font must be defined
\usepackage{amsthm}% must be loaded before newtxmath
\usepackage[libertine,cmintegrals,bigdelims,vvarbb]{newtxmath}
\usepackage[scr=rsfso]{mathalfa}
\usepackage{bm}% load after all math to give access to bold math
%% After loading math package, switch to osf in text.
\useosf % for osf in normal text
